# ============================================================
# 新 VM：重合并 6 个 benchmark + 重评分（不重生成）
# ============================================================
# 根因已定位：generate_offline_vllm.py 里 --model_path/--data_path 被标成 required=True，
# 而 --merge 模式不传这两个参数，导致 argparse 在进 merge 分支前就报错退出，
# merge_shards 从未被调用 → 6 个 merged parquet 全没生成（commit 已修复）。
# 本文件：git pull 拿到修复 → 重合并 6 个 benchmark（几秒）→ 重评分（几秒）。
# 完全不碰生成，不重跑 vLLM，省下 ~1.5h。
# ============================================================

cd /inspire/hdd/project/multi-agent/zhangweinan-24046/dk/CacheOPD/eopd-baseline
git pull                                  # 拿到修复后的 generate_offline_vllm.py
git log --oneline -1                      # 应看到修复 commit

DATA_DIR=/inspire/hdd/project/multi-agent/zhangweinan-24046/dk/data
PY=/opt/conda/envs/eopd/bin/python3
BENCH=(MATH500 AMC23 Minerva OlympiadBench AIME24 AIME25)

# 1) 重合并 6 个 benchmark（修复后的 --merge 不再要求 model_path/data_path）
for B in "${BENCH[@]}"; do
  echo "==================== merge $B ===================="
  $PY examples/on_policy_distillation/generate_offline_vllm.py --merge \
    --input_glob "$DATA_DIR/eval_gen/$B.parquet.shard*.parquet" \
    --output_path "$DATA_DIR/eval_gen/$B.parquet"
done

# 2) 确认 6 个 merged 都生成了（应看到 6 个不带 shard 的 .parquet）
ls -la $DATA_DIR/eval_gen/*.parquet | grep -v shard

# 3) 评分（直接读已有的 merged parquet，写 actor_huggingface_scores.{txt,json}）
$PY examples/on_policy_distillation/score_avg_pass_at_k.py \
  --output $DATA_DIR/eval_results/actor_huggingface_scores.json \
  --input MATH500=$DATA_DIR/eval_gen/MATH500.parquet \
  --input AMC23=$DATA_DIR/eval_gen/AMC23.parquet \
  --input Minerva=$DATA_DIR/eval_gen/Minerva.parquet \
  --input OlympiadBench=$DATA_DIR/eval_gen/OlympiadBench.parquet \
  --input AIME24=$DATA_DIR/eval_gen/AIME24.parquet \
  --input AIME25=$DATA_DIR/eval_gen/AIME25.parquet \
  | tee $DATA_DIR/eval_results/actor_huggingface_scores.txt

echo "==================== 结果 ===================="
tail -12 $DATA_DIR/eval_results/actor_huggingface_scores.txt
